マクロサンプル

一番左の列のセルに行番号を設定する

document.dataTable.selectFirstColumn();         // 一番左側のセルを選択
document.dataTable.selectEntireColumn();        // 選択範囲を縦方向に拡張
document.dataTable.processSelectedCell(function(row, column){
	document.dataTable.setValue(row + 1, row, column);    // 選択範囲の各セルに行番号+1をセットする
});

値が空でないセルをすべて選択する

document.dataTable.processAllCell(function(row, column){
	if(document.dataTable.getValue(row, column) != ''){
		document.dataTable.select(row, column, true);
	}
});

選択中のセルの値を、右側のタブの同じ位置に貼り付ける

document.dataTable.copy();        // セルの値をクリップボードにコピーする
var row = document.dataTable.getFocusedRow();    // 選択位置を取得
var column = document.dataTable.getFocusedColumn();
document.nextTab();        // 右側のセルを選択
document.dataTable.select(row, column);    // セルを選択
document.dataTable.paste();        // クリップボードの内容を貼り付ける
